#include <iostream>
#include <stack>
using namespace std;

int main ()
{
    stack<int> s;

    s.push(1);
    s.emplace(3);
    
    while(!s.empty()) {
        cout<<s.top()<<",";
        s.pop();
    }
    cout<<endl;
}
// 3,1
