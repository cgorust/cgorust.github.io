array method in C++

    * create: array\<Type,Size> a{};
* add (N/A)
* change
    * swap: swap()
    * fill: fill()
* query
    * at: at(), get(), operator[]
    * front: front()
    * back: back()
    * size: size(), max_size()
    * empty: empty()
* delete (N/A)
* iterator
```
#include <iostream>
#include <array>
using namespace std;

int main() {
    array<int, 10> a{};
    array<int, 10> b{};
    
    b.fill(1);
    a.swap(b);
    
    cout<<a[2]<<","<<get<2>(a)<<","<<a[2]<<endl;
    cout<<a.front()<<","<<a.back()<<","<<a.size()<<","<<a.max_size()
        <<","<<a.empty()<<endl;    
 
   for(auto &i: a) 
        cout<<i<<",";
}
```
