set method in C++

* create
    * constructor, copy constructor, copy assignment operator
```
#include <iostream>
#include <set>
using namespace std;

struct classComp {
  bool operator()(int l, int r) { return l<r; }
};
bool funComp(int l, int r) { return l<r; }

int main () {
    set<int> s1;  // default constructor
    set<int> s2(s1);  // copy constructor
    s2=s1;  // assignment operator

    int a[]= {1,2,3,4,5};
    set<int> s3(a, a+5);  // range of pointer

    set<int> s4(s3.begin(), s3.end());  // iterator

    std::set<int, classComp> s5;    // with class comparator

    std::set<int, decltype(&funComp)> s6(&funComp);  // with function compartor
}
```
* iterator: An iterator for a set is bidirectional, so it does not support +(int). The only supported move operations are ++ and --.
    * begin: Return iterator to beginning
    * end: Return iterator to end (public member functio
    * rbegin: Return reverse iterator to reverse beginning
    * rend: Return reverse iterator to reverse end (public 
    * cbegin: Return const_iterator to beginning
    * cend: Return const_iterator to end
    * crbegin: Return const_reverse_iterator to reverse beginning
    * crend: Return const_reverse_iterator to reverse end
```
#include <iostream>
#include <set>
using namespace std;

int main (){
  set<int> s{1,2,3,4,5};

  for (auto it=s.begin(); it!=s.end(); ++it)
    cout<<*it<<",";
  cout<<endl;
  for (auto it=s.cbegin(); it!=s.cend(); ++it)
    cout<<*it<<",";
  cout<<endl;
  for (auto it=s.rbegin(); it!=s.rend(); ++it)
    cout<<*it<<",";
  cout<<endl;
  for (auto it=s.crbegin(); it!=s.crend(); ++it)
    cout<<*it<<",";
  cout<<endl;
}

* change
    * swap: Swap the contents of one vector with another vector of same type. Sizes may differ.
```
#include <iostream>
#include <set>
using namespace std;

int main() {
    int a[]={1,2,3,4,5,6};
    set<int> s1(a,a+3);
    set<int> s2(a+3,a+6);

    s1.swap(s2);

    for(auto &i:s1)
        cout<<i<<",";
    cout<<endl;
    
    for(auto &i:s2)
        cout<<i<<",";
    cout<<endl;
}
/*
4,5,6,
1,2,3,
*/

* query
    * value
        * find: Get element.
        * count: Count elements with a specific value.
        * lower_bound: Return lower bound.
        * upper_bould: Return upper bound.
        * equal_range: Get range of equal elements.
    * set
        * empty: Return whether the container is empty.
        * size: Return the number of elements in the container.
        * max_size: Returns the maximum number of elements that the container can hold.
```
int main() {
    set<int> s{1,2,3,4,5,6};

    auto it=s.find(2);
    if(it!=s.end()) cout<<"found"<<endl;
    cout<<s.count(2)<<endl;
    
    auto itl = s.lower_bound(3);
    it=s.begin();
    while(it!=itl){
        cout<<*it++<<",";
    }
    cout<<endl;

    it=s.upper_bound(3);
    while(it!=s.end()) {
        cout<<*it++<<",";
    } 
    cout<<endl;

    auto r=s.equal_range(3);
    for(auto it=r.first; it!=r.second; it++)
        cout<<*it<<",";
    cout<<endl;
    
    cout<<s.empty()<<endl;
    cout<<s.size()<<endl;
    cout<<s.max_size()<<endl;
}
/*

1,2,
4,5,6,
3,
0
6
230584300921369395

* delete 
    * erase: Remove a element from a container.
    * clear: Remove all the elements of the container.
```
#include <iostream>
#include <set>
using namespace std;

int main ()
{
    set<int> s{1,2,3,4,5};

    auto it = s.begin();
    s.erase(it);
    s.erase(4);

    for(auto &i:s)
        cout<<i<<",";
    cout<<endl;
    s.clear();
}
//2,3,5
