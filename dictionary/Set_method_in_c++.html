```
#include <iostream>
#include <set>
using namespace std;

struct classComp {
  bool operator()(int l, int r) { return l<r; }
};
bool funComp(int l, int r) { return l<r; }

int main () {
    set<int> s1;  // default constructor
    set<int> s2(s1);  // copy constructor

    int a[]= {1,2,3,4,5};
    set<int> s3(a, a+5);  // range of pointer
  
    set<int> s4(s3.begin(), s3.end());  // iterator

    std::set<int, classComp> s5;    // with class comparator
  
    std::set<int, decltype(&funComp)> s6(&funComp);  // with function compartor
}
```
