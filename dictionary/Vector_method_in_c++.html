vector method in C++

* create
    * create: vector\<type> v{}
```
#include <vector>

int main() {
    vector<int> v{1,2,3,4,5};
}
```
* add
    * push_back: push_back()
    * insert: 
        * insert(): Insert takes (a reference to) an object.
        * emplace(): Emplace takes the arguments necessary to construct an object in place
```
#include <vector>
#include <iostream>
using namespace std;

struct Foo
{
    Foo(int n, double x) {}
};

int main() {
    vector<int> v{1,2,3,4,5};
    v.push_back(1);
    v.insert(v.begin()+2,4);
    
    for(auto &i:v) 
        cout<<i<<",";
    cout<<endl;

    std::vector<Foo> f;
    f.emplace(f.begin(), 42, 3.1416);
    f.insert(f.begin(), Foo(42, 3.1416));
}
```
* change
    * swap: swap()
    * assign: assign()
    * capacity: resize(), shrink_to_fit(), reserve()
```
#include <vector>
#include <iostream>
using namespace std;

int main() {
    vector<int> v{};
    vector<int> w{1,2,3,4,5};

    v.assign(10, 11);
    v.swap(w);
    for(auto &i: v)
        cout<<i<<",";
    cout<<endl;
    
    cout<<v.size()<<",";
    v.resize(100);
    cout<<v.size()<<",";
    v.shrink_to_fit();
    cout<<v.size()<<",";
    v.reserve(6);
    cout<<v.size()<<",";
}
```
* query
    * at: operator[], at(), data()
    * front: front()
    * back: back()
    * size: size()
    * max_size: max_size()
    * capacity: capacity()
    * empty: empty()
```
#include <vector>
#include <iostream>
using namespace std;

int main() {
    vector<int> v{1,2,3,4,5};

    cout<<v[1]<<","<<v.at(1)<<","<<*v.data()+1<<endl;
    cout<<v.front()<<","<<v.back()<<","<<v.size()<<","
        <<v.max_size()<<","<<v.capacity()<<","<<v.empty()<<endl;
}
```
* delete
    * pop_back: pop_back()
    * erase: erase()
    * clear: clear()
```
#include <vector>
using namespace std;

int main() {
    vector<int> v{1,2,3,4,5};

    v.pop_back();
    v.erase(v.begin()+2);
    v.clear();
}

* iterator
    * begin(), end(); rbegin(), rend(); cbegin(), cend(); crbegin(), crend();
```
#include <iostream>
#include <vector>
using namespace std;
 
int main() {
    vector<int> v{1,2,3,4,5};
 
    for (auto i = v.begin(); i != v.end(); ++i)
        cout << *i << " ";
    cout<<endl;    
    for (auto i = v.cbegin(); i != v.cend(); ++i)
        cout << *i << " ";
    cout<<endl;    
    for (auto i = v.rbegin(); i != v.rend(); ++i)
        cout << *i << " ";
    cout<<endl;    
    for (auto i = v.crbegin(); i != v.crend(); ++i)
        cout << *i << " ";
    cout<<endl;    
}
