#include <iostream>
using namespace std;

class Stock {
public:
    virtual void accept(class Visitor *) = 0;
};
  
class Apple : public Stock {
  public:
    void buy() { cout << "Apple::buy\n"; }
    void sell() { cout << "Apple::sell\n"; }
    void accept(Visitor *);
};
  
class Google : public Stock {
  public:
    void buy() { cout << "Google::buy\n"; }
    void sell() { cout << "Google::sell\n"; }
    void accept(Visitor *);
};

class Visitor {
public:
    virtual void visit(Apple *) = 0;
    virtual void visit(Google *) = 0;

    void total_stocks() {
        cout << "m_num_apple " << m_num_apple 
             << ", m_num_google " << m_num_google << '\n';
    }
protected:
    static int m_num_apple, m_num_google;
};
int Visitor::m_num_apple = 0;
int Visitor::m_num_google = 0;

void Apple::accept(Visitor *v) { v->visit(this); }
  
void Google::accept(Visitor *v) { v->visit(this); }

class BuyVisitor : public Visitor {
  public:
    BuyVisitor() { m_num_apple = m_num_google = 0; }
    void visit(Apple *r) {
        ++m_num_apple;
        r->buy();
    }
    void visit(Google *b) {
        ++m_num_google;
        b->buy();
    }
};
  
class SellVisitor : public Visitor {
  public:
    void visit(Apple *a) {
        --m_num_apple;
        a->sell();
    }
    void visit(Google *g) {
        --m_num_google;
        g->sell();
    }
};
   
int main() {
    Stock *set[] = { new Apple, new Google, new Google,
                     new Apple, new Apple, 0};
  
    BuyVisitor buy_operation;
    for (int i = 0; set[i]; i++) {
        set[i]->accept(&buy_operation);
    }
    buy_operation.total_stocks();
  
    SellVisitor sell_operation;
    for (int i = 0; set[i]; i++) {
        set[i]->accept(&sell_operation);
    }
    sell_operation.total_stocks();
}
